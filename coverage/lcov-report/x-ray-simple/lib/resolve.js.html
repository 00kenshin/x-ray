function select($, selector) {
  if ($.is && $.is(selector)) return $;
  return $.find ? $.find(selector) : $(selector);
}
 
function attribute($el, attr) {
  switch(attr) {
    case 'html':
      return $el.html();
    case 'text':
      return $el.text();
    default:
      return $el.attr(attr);
  }
}
